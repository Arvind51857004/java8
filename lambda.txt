package java8;


@FunctionalInterface
public interface Funinter {     ///functionalinterface should consist of only one abstract method
	
	public abstract int easy();
	
	default int get() {
		return 100;
		}
	
	default void normal() {
		System.out.print("from normal method");
	}
	
	static void set() {
		System.out.println("static method");
	}

}
